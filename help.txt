1. docker installation (apt version)
- https://docs.docker.com/engine/install/ubuntu/
check if group exists
- sudo getent group docker 
add username to docker group
sudo usermod -aG docker username

docker build --rm --tag rkbvn89/qemu_x86_64_yocto:onlytest -f Dockerfile .
docker run rkbvn89/qemu_x86_64_yocto:v1
docker push rkbvn89/qemu_x86_64_yocto:v1
docker pull rkbvn89/qemu_x86_64_yocto:v1
docker run --rm -it rkbvn89/qemu_x86_64_yocto:onlytest

2. qemu-system-x86 
- sudo apt install qemu-system-x86

qemu-system-x86_64 -kernel bzImage-qemux86-64.bin -drive format=raw,file=core-image-minimal-qemux86-64.ext4 -append "root=/dev/hda console=ttyS0" -nographic
